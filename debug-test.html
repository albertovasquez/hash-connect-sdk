Debug Commands

Run these in your browser console to inspect the SDK state:

// Check if SDK is loaded
console.log('SDK loaded:', !!window.HASHConnect);

// Check SDK methods
console.log('SDK methods:', window.HASHConnect ? Object.keys(window.HASHConnect) : 'N/A');

// Check if SDK is ready
console.log('SDK ready:', window.HASHConnect?.isReady());

// Get current user
console.log('Current user:', window.HASHConnect?.getUser());

// Check session storage
console.log('Session ID:', localStorage.getItem('hc:sessionId'));
console.log('Address:', localStorage.getItem('hc:address'));
console.log('Access Token:', localStorage.getItem('hc:accessToken') ? 'Present' : 'None');

// Check DOM elements
console.log('Modal exists:', !!document.getElementById('hash-connect-modal'));
console.log('QR div exists:', !!document.getElementById('hash-connect-qrcode'));
